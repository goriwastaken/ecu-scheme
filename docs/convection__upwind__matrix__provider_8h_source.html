132 template <typename SCALAR, typename FUNCTOR>
133 Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic>
134 ConvectionUpwindMatrixProvider<SCALAR, FUNCTOR>::Eval(
135  const lf::mesh::Entity& entity) {
136  LF_ASSERT_MSG(entity.RefEl() == lf::base::RefEl::kTria(),
137  "Function only defined for triangular cells");
138  const lf::geometry::Geometry* geo_ptr = entity.Geometry();
139  const Eigen::MatrixXd corners = lf::geometry::Corners(*geo_ptr);
140  const double area = lf::geometry::Volume(*geo_ptr);
141  LF_ASSERT_MSG(area > 0, "Area of cell must be positive");
142 
143  const size_t num_local_dofs = fe_space_->LocGlobMap().NumLocalDofs(entity);
144  LF_ASSERT_MSG(num_local_dofs == 3 || num_local_dofs == 6,
145  "Class supports only linear and quadratic FE spaces");
146  // std::cout << "num local dofs: " << num_local_dofs << '\n';
147  Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic> element_matrix(
148  num_local_dofs, num_local_dofs);
149 
150  // Compute normals of vertices of the triangular cell
151  Eigen::Matrix3d X;
152  X.col(0) = Eigen::Vector3d::Ones();
153  X.rightCols(2) = corners.transpose();
154  Eigen::MatrixXd temporary_gradients = X.inverse().bottomRows(2);
155  Eigen::MatrixXd vertex_normals = -temporary_gradients;
156 
157  // Compute signed area of the cell for checking the orientation of vertices
158  const double signed_area =
159  (corners(0, 1) - corners(0, 0)) * (corners(1, 2) - corners(1, 0)) -
160  (corners(0, 2) - corners(0, 0)) * (corners(1, 1) - corners(1, 0));
161  // Set orientation based on signed area
162  const bool is_clockwise = (signed_area < 0);
163 
164  // Compute edge midpoint coordinates - only needed for quadratic FE space
165  Eigen::MatrixXd midpoints(2, 3);
166  for (int i = 0; i < 3; ++i) {
167  midpoints.col(i) = 0.5 * (corners.col((i + 1) % 3) + corners.col(i));
168  }
169 
170  // Prepare matrices for the computation of the element matrix and separate
171  // implementation for linear and quadratic FE space
172  Eigen::MatrixXd velocities(2, num_local_dofs);
173  Eigen::MatrixXd all_nodes(2, num_local_dofs);
174  std::vector<double> local_masses;
175 
176  // Start case distinction linear or quadratic FE space
177  if (num_local_dofs == 3) {
178  // Linear FE space case
179  all_nodes << corners;
180  velocities << v_(all_nodes.col(0)), v_(all_nodes.col(1)),
181  v_(all_nodes.col(2));
182  for (const lf::mesh::Entity* e : entity.SubEntities(2)) {
183  local_masses.push_back(masses_vertices_(*e));
184  }
185  LF_ASSERT_MSG(local_masses.size() == 3,
186  "There must be three masses, one for each basis function");
187  // the gradients in the linear case are obtained as the vertex normals, just
188  // with opposite sign
189  Eigen::MatrixXd gradients_linearFE = -vertex_normals;
190 
191  for (int l = 0; l < 3; ++l) {
192  Eigen::Vector2d vl = velocities.col(l);
193  // Vertex a^l is upwind iff product of v(a^l) with both adjacent normals
194  // is positive
195  if (vl.dot(vertex_normals.col((l + 2) % 3)) >= 0 &&
196  vl.dot(vertex_normals.col((l + 1) % 3)) >= 0) {
197  // a^l is upwind
198  element_matrix.row(l) =
199  local_masses[l] * vl.transpose() * gradients_linearFE;
200  } else {
201  // a^l is not upwind
202  element_matrix.row(l) = Eigen::Vector<SCALAR, 3>::Zero();
203  }
204  }
205  // End of linear FE space case
206  } else {
207  // Quadratic FE space
208  all_nodes << corners, midpoints;
209  velocities << v_(all_nodes.col(0)), v_(all_nodes.col(1)),
210  v_(all_nodes.col(2)), v_(all_nodes.col(3)), v_(all_nodes.col(4)),
211  v_(all_nodes.col(5));
212 
213  for (const lf::mesh::Entity* e : entity.SubEntities(2)) {
214  local_masses.push_back(masses_vertices_(*e));
215  }
216  for (const lf::mesh::Entity* e : entity.SubEntities(1)) {
217  local_masses.push_back(masses_edges_(*e));
218  }
219  // std::cout << "local masses size: " << local_masses.size() << std::endl;
220  // std::cout << "local masses: " << local_masses[0] << ", " <<
221  // local_masses[1] << ", " << local_masses[2] << ", " << local_masses[3] <<
222  // ", " << local_masses[4] << ", " << local_masses[5] << std::endl;
223  LF_ASSERT_MSG(local_masses.size() == 6,
224  "There must be six masses, one for each basis function");
225 
226  // Row-vector of barycentric coordinate functions based on global
227  // coordinates of nodes
228  auto bary_functions =
229  [area, corners, is_clockwise](
230  const Eigen::Vector2d& xh) -> Eigen::Matrix<double, 1, 3> {
231  Eigen::Matrix<double, 1, 3> bary;
232  const double coeff = 1.0 / (2.0 * area);
233  // barycentric function 1
234  bary.col(0) = coeff * (xh - corners.col(1)).transpose() *
235  (Eigen::Vector2d(2, 1) << corners(1, 1) - corners(1, 2),
236  corners(0, 2) - corners(0, 1))
237  .finished();
238  // bary 2
239  bary.col(1) = coeff * (xh - corners.col(2)).transpose() *
240  (Eigen::Vector2d(2, 1) << corners(1, 2) - corners(1, 0),
241  corners(0, 0) - corners(0, 2))
242  .finished();
243  // bary 3
244  bary.col(2) = coeff * (xh - corners.col(0)).transpose() *
245  (Eigen::Vector2d(2, 1) << corners(1, 0) - corners(1, 1),
246  corners(0, 1) - corners(0, 0))
247  .finished();
248  if (is_clockwise) {
249  return -bary;
250  }
251  return bary;
252  };
253 
254  // Matrix of gradients of barycentric coordinate functions based on global
255  // coordinates of nodes
256  auto bary_functions_grad =
257  [area, corners, is_clockwise](
258  const Eigen::Vector2d& xh) -> Eigen::Matrix<double, 2, 3> {
259  Eigen::Matrix<double, 2, 3> bary_grad;
260  const double coeff = 1.0 / (2.0 * area);
261  // bary 1
262  bary_grad.col(0) =
263  coeff * (Eigen::Vector2d(2, 1) << corners(1, 1) - corners(1, 2),
264  corners(0, 2) - corners(0, 1))
265  .finished();
266  // bary 2
267  bary_grad.col(1) =
268  coeff * (Eigen::Vector2d(2, 1) << corners(1, 2) - corners(1, 0),
269  corners(0, 0) - corners(0, 2))
270  .finished();
271  // bary 3
272  bary_grad.col(2) =
273  coeff * (Eigen::Vector2d(2, 1) << corners(1, 0) - corners(1, 1),
274  corners(0, 1) - corners(0, 0))
275  .finished();
276  if (is_clockwise) {
277  return -bary_grad;
278  }
279  return bary_grad;
280  };
281 
282  // Matrix of gradients of local shape functions based on global coordinates
283  // of nodes in quadratic Lagrangian FE space
284  auto gradientsLocalShapeFunctions =
285  [bary_functions, bary_functions_grad](
286  const Eigen::Vector2d xh) -> Eigen::Matrix<double, 2, 6> {
287  Eigen::Matrix<double, 2, 6> gradients;
288  // barycentric coordinate functions
289  Eigen::Matrix<double, 1, 3> temp = bary_functions(xh);
290  // Eigen::Matrix<double, 1, 3> temp = bary_new(xh);
291  Eigen::Matrix<double, 2, 3> grads_bary = bary_functions_grad(xh);
292  Eigen::RowVector3d l;
293  l << temp(0, 0), temp(0, 1), temp(0, 2);
294 
295  gradients.col(0) = grads_bary.col(0) * (4 * l[0] - 1);
296  gradients.col(1) = grads_bary.col(1) * (4 * l[1] - 1);
297  gradients.col(2) = grads_bary.col(2) * (4 * l[2] - 1);
298  gradients.col(3) =
299  4 * (grads_bary.col(0) * l[1] + grads_bary.col(1) * l[0]);
300  gradients.col(4) =
301  4 * (grads_bary.col(1) * l[2] + grads_bary.col(2) * l[1]);
302  gradients.col(5) =
303  4 * (grads_bary.col(0) * l[2] + grads_bary.col(2) * l[0]);
304  return gradients;
305  };
306 
307  // Main loop for the computation of the element matrix
308  for (int l = 0; l < 6; ++l) {
309  Eigen::Vector2d vl = velocities.col(l);
310  // compute product of v(a^l) with gradients of basis function of a^l
311  Eigen::Matrix<double, 2, 6> grads =
312  gradientsLocalShapeFunctions(all_nodes.col(l));
313  Eigen::Matrix<double, 1, 6> contribution = vl.transpose() * grads;
314  // Eigen::Matrix<double, 1, 6> shapeFunctionsContribution =
315  // localShapeFunctions(all_nodes.col(l)); Eigen::Matrix<double, 1, 6>
316  // totalFunctionContribution =
317  // contribution.cwiseProduct(shapeFunctionsContribution);
318 
319  // Vertex a^l is upwind iff product of v(a^l) with both adjacent normals
320  // is positive
321  if (l < 3) {
322  // first 3 nodes are vertices of triangle
323  if (vl.dot(vertex_normals.col((l + 2) % 3)) >= 0 &&
324  vl.dot(vertex_normals.col((l + 1) % 3)) >= 0) {
325  // a^l is upwind
326  element_matrix.row(l) = local_masses[l] * contribution;
327  LF_ASSERT_MSG(element_matrix.row(l).norm() == 0,
328  "Vertex masses should be 0 in this case");
329  } else {
330  // a^l is not upwind
331  element_matrix.row(l) = Eigen::Vector<SCALAR, 6>::Zero();
332  }
333  } else {
334  // last 3 nodes are edge midpoints of triangle
335  const Eigen::Matrix<double, 2, 3> outward_normals =
336  computeOutwardNormalsTria(entity);
337  // Midpoint m^l is upwind iff product of v(m^l) with corresponding
338  // outward normals is positive
339  if (vl.dot(outward_normals.col(l % 3)) >= 0) {
340  // m^l is upwind
341  if (vl.dot(outward_normals.col(l % 3)) == 0) {
342  element_matrix.row(l) = 0.5 * local_masses[l] * contribution;
343  } else {
344  element_matrix.row(l) = local_masses[l] * contribution;
345  }
346 
347  } else {
348  // m^l is not upwind
349  element_matrix.row(l) = Eigen::Vector<SCALAR, 6>::Zero();
350  }
351  }
352  }
353  // End of quadratic FE space case
354  } // End case distinction linear or quadratic FE space
355 
356  return element_matrix;
357 }
