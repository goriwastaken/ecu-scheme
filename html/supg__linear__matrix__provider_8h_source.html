53 template <typename FUNCTOR>
54 Eigen::Matrix3d SupgConvectiveEMP<FUNCTOR>::Eval(
55  const lf::mesh::Entity& entity) {
56  LF_ASSERT_MSG(lf::base::RefEl::kTria() == entity.RefEl(),
57  "Function only defined for triangular cells");
58 
59  const lf::geometry::Geometry* geo_ptr = entity.Geometry();
60  Eigen::Matrix3d loc_mat;
61 
62  const Eigen::MatrixXd corners = lf::geometry::Corners(*geo_ptr);
63 
64  // calculate the gradients of the basis functions.
65  // See \lref{cpp:gradbarycoords}, \lref{mc:ElementMatrixLaplLFE} for details.
66  Eigen::Matrix3d grad_helper;
67  grad_helper.col(0) = Eigen::Vector3d::Ones();
68  grad_helper.rightCols(2) = corners.transpose();
69  // Matrix with gradients of the local shape functions in its columns
70  const Eigen::MatrixXd grad_basis = grad_helper.inverse().bottomRows(2);
71 
72  // evaluate velocity field at the vertices:
73  Eigen::MatrixXd velocities(2, 3);
74  velocities << v_(corners.col(0)), v_(corners.col(1)), v_(corners.col(2));
75 
76  // compute local matrix using local trapezoidal rule:
77  loc_mat = lf::geometry::Volume(*geo_ptr) / 3.0 * velocities.transpose() *
78  grad_basis;
79 
80  return loc_mat;
81 }

113 template <typename FUNCTOR_EPS, typename FUNCTOR_V, typename FUNCTOR_F>
114 Eigen::Matrix3d SupgStabilizationEMP<FUNCTOR_EPS, FUNCTOR_V, FUNCTOR_F>::Eval(
115  const lf::mesh::Entity& entity) {
116  LF_ASSERT_MSG(lf::base::RefEl::kTria() == entity.RefEl(),
117  "Function only defined for triangular cells");
118  const lf::geometry::Geometry* geo_ptr = entity.Geometry();
119  Eigen::Matrix3d loc_mat;
120  loc_mat.setZero();
121 
122  const Eigen::MatrixXd corners = lf::geometry::Corners(*geo_ptr);
123 
124  // calculate the gradients of the basis functions.
125  // See \lref{cpp:gradbarycoords}, \lref{mc:ElementMatrixLaplLFE} for details.
126  Eigen::Matrix3d grad_helper;
127  grad_helper.col(0) = Eigen::Vector3d::Ones();
128  grad_helper.rightCols(2) = corners.transpose();
129  // Matrix with gradients of the local shape functions in its columns
130  const Eigen::MatrixXd grad_basis = grad_helper.inverse().bottomRows(2);
131 
132  // Local control parameter
133  const double delta = ecu_scheme::assemble::Delta(entity, eps_(corners.col(0)),
134  v_(corners.col(0)));
135 
136  // weight for the local trapezoidal rule:
137  const double weight = delta * lf::geometry::Volume(*geo_ptr) / 3.0;
138 
139  // compute local matrix based on local trapezoidal rule:
140  for (int i = 0; i < 3; ++i) {
141  Eigen::Vector2d corner = corners.col(i);
142  Eigen::MatrixXd trial_eval =
143  v_(corner).transpose() * grad_basis -
144  f_(corner) * Eigen::Vector3d::Ones().transpose();
145  Eigen::MatrixXd test_eval = v_(corner).transpose() * grad_basis;
146 
147  loc_mat += weight * test_eval.transpose() * trial_eval;
148  }
149 
150  return loc_mat;
151 }

167 template <typename DIFFUSION_COEFF, typename CONVECTION_COEFF,
168  typename FUNCTOR_F, typename FUNCTOR_G>
169 Eigen::VectorXd SolveCDBVPSupg(
170  const std::shared_ptr<lf::uscalfe::FeSpaceLagrangeO1<double>>& fe_space,
171  DIFFUSION_COEFF eps, CONVECTION_COEFF v, FUNCTOR_F f, FUNCTOR_G g) {
172  // Wrap functions into mesh functions
173  lf::mesh::utils::MeshFunctionGlobal mf_g{g};
174  lf::mesh::utils::MeshFunctionGlobal mf_eps{eps};
175  lf::mesh::utils::MeshFunctionGlobal mf_f{f};
176 
177  // mesh and dofhanlder
178  auto mesh_p = fe_space->Mesh();
179  const lf::assemble::DofHandler& dofh{fe_space->LocGlobMap()};
180 
181  // Matrix in triplet format holding Galerkin matrix, zero initially.
182  lf::assemble::COOMatrix<double> A(dofh.NumDofs(), dofh.NumDofs());
183 
184  // ASSEMBLE GALERKIN MATRIX
185  // First the part corresponding to the laplacian
186  lf::uscalfe::ReactionDiffusionElementMatrixProvider laplacian_provider(
187  fe_space, mf_eps, lf::mesh::utils::MeshFunctionConstant(0.0));
188  lf::assemble::AssembleMatrixLocally(0, dofh, dofh, laplacian_provider, A);
189 
190  // Next part corresponding to the convection term:
191  ecu_scheme::assemble::SupgConvectiveEMP<decltype(v)> convection_provider(v);
192  lf::assemble::AssembleMatrixLocally(0, dofh, dofh, convection_provider, A);
193 
194  // Stabilization term:
195  ecu_scheme::assemble::SupgStabilizationEMP<decltype(eps), decltype(v),
196  decltype(f)>
197  stabilization_provider(eps, v, f);
198  lf::assemble::AssembleMatrixLocally(0, dofh, dofh, stabilization_provider, A);
199 
200  // RIGHT-HAND SIDE VECTOR
201  Eigen::VectorXd phi(dofh.NumDofs());
202  phi.setZero();
203  lf::uscalfe::ScalarLoadElementVectorProvider elvec_provider(fe_space, mf_f);
204  lf::assemble::AssembleVectorLocally(0, dofh, elvec_provider, phi);
205  // IMPOSE DIRICHLET BC
206  // Obtain specification for shape functions on edges
207  const lf::fe::ScalarReferenceFiniteElement<double>* rsf_edge_p =
208  fe_space->ShapeFunctionLayout(lf::base::RefEl::kSegment());
209 
210  // // Create a dataset of boolean flags indicating edges on the boundary of
211  // the
212  // // mesh
213  auto bd_flags{lf::mesh::utils::flagEntitiesOnBoundary(mesh_p, 1)};
214 
215  // Fetch flags and values for degrees of freedom located on Dirichlet
216  // boundary.
217  auto ess_bdc_flags_values{
218  lf::fe::InitEssentialConditionFromFunction(*fe_space, bd_flags, mf_g)};
219 
220  // Eliminate Dirichlet dofs from linear system
221  lf::assemble::FixFlaggedSolutionComponents<double>(
222  [&ess_bdc_flags_values](lf::uscalfe::glb_idx_t gdof_idx) {
223  return ess_bdc_flags_values[gdof_idx];
224  },
225  A, phi);
226 
227  // SOLVE LINEAR SYSTEM
228  Eigen::SparseMatrix A_crs = A.makeSparse();
229  Eigen::SparseLU<Eigen::SparseMatrix<double>> solver;
230  solver.compute(A_crs);
231  Eigen::VectorXd sol_vec = solver.solve(phi);
232 
233  return sol_vec;
234 }
