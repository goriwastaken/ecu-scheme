6 Eigen::Matrix<double, 2, 3> computeOutwardNormals(
7  const lf::mesh::Entity& entity) {
8  Eigen::Matrix<double, 2, 3> normals;
9  const lf::geometry::Geometry* geo_ptr = entity.Geometry();
10  const Eigen::MatrixXd corners = lf::geometry::Corners(*geo_ptr);
11 
12  Eigen::Matrix<double, 3, 3> delta;
13  delta.block<3, 2>(0, 0) = corners.transpose();
14  delta.block<3, 1>(0, 2) = Eigen::Vector3d::Ones();
15  const double det = delta.determinant();
16  // rotation matrix of angle pi/2 is: 0, -1,
17  // 1, 0
18  const Eigen::Matrix<double, 2, 2> rotation_matrix =
19  (Eigen::Matrix<double, 2, 2>() << 0, -1, 1, 0).finished();
20  for (int i = 0; i < 3; ++i) {
21  const Eigen::Vector2d edge = corners.col((i + 1) % 3) - corners.col(i);
22  normals.col(i) = rotation_matrix * edge;
23  normals.col(i) /= normals.col(i).norm();
24  }
25  if (det > 0) {
26  // flip sign
27  normals *= -1;
28  }
29 
30  return normals;
31 }

33 Eigen::Matrix<double, 2, 3> computeTangentialComponents(
34  const lf::mesh::Entity& entity) {
35  const Eigen::Matrix<double, 2, 3> tria_normals =
36  computeOutwardNormals(entity);
37  Eigen::Matrix<double, 2, 3> tangential_components;
38  tangential_components << -tria_normals.row(1), tria_normals.row(0);
39  return tangential_components;

42 void EnforceInflowBDCOneform(
43  const std::shared_ptr<lf::uscalfe::UniformScalarFESpace<double>>& fe_space,
44  lf::assemble::COOMatrix<double>& A, Eigen::VectorXd& phi,
45  std::function<
46  Eigen::Matrix<double, 2, 1>(const Eigen::Matrix<double, 2, 1, 0>&)>
47  dirichlet) {
48  // Generate DOFHandler object corresponding to edge element basis functions
49  auto mesh_p = fe_space->Mesh();
50  const lf::assemble::DofHandler& dofh_edge = lf::assemble::UniformFEDofHandler(
51  mesh_p, {{lf::base::RefEl::kSegment(), 1}});
52  lf::mesh::utils::AllCodimMeshDataSet<bool> bd_flags(mesh_p, false);
53  // tolerance for geometric tests
54  const double kTol = 1e-8;
55 
56  // Loop over all edges
57  for (const auto& edge : mesh_p->Entities(1)) {
58  LF_ASSERT_MSG(edge->RefEl() == lf::base::RefEl::kSegment(),
59  "Entity should be an edge");
60  const lf::geometry::Geometry* geo_ptr = edge->Geometry();
61  const Eigen::MatrixXd corners = lf::geometry::Corners(*geo_ptr);
62  // Check if the edge lies on $\Gamma_{\mathrm{in}}$ (geometric test)
63  if ((corners(0, 0) + corners(0, 1)) / 2. > 1. - kTol ||
64  (corners(1, 0) + corners(1, 1)) / 2. < kTol) {
65  // Add the edge to the flagged entities
66  bd_flags(*edge) = true;
67  }
68  }
69  // Set boundary condition data to flagged edges
70  // remember initEssentialConditionFromFunction() doesn't work if underlying FE
71  // space remains the lagrangian one
72  // auto selector = [&](lf::base::size_type idx) -> std::pair<bool, double> {
73  // const auto &e = dofh.Entity(idx);
74  // return {e.RefEl() == lf::base::RefElType::kSegment() && bd_flags(*e),
75  // 0};
76  // };
77 
78  std::vector<std::pair<bool, double>> ess_dof_select(dofh_edge.NumDofs(),
79  {false, 0.0});
80  // We want to enforce inflow BDC at tangential components of the edges
81  // One way to do this is iterate through all cells and stop at marked edges
82  // This allows to precompute the outward normals, and thus tangential
83  // components beforehand
84  for (const lf::mesh::Entity* entity : mesh_p->Entities(0)) {
85  const Eigen::Matrix<double, 2, 3> tangential_components =
86  computeTangentialComponents(*entity);
87  unsigned int edge_idx = 0;
88  for (const lf::mesh::Entity* edge : entity->SubEntities(1)) {
89  // indexing is consistent with the one of the edge DOF handler
90  const lf::geometry::Geometry* geo_ptr = edge->Geometry();
91  const Eigen::MatrixXd endpoints = lf::geometry::Corners(*geo_ptr);
92  if (bd_flags(*edge)) {
93  const Eigen::Vector2d eval_bdc_at_midpoint =
94  dirichlet((endpoints.col(0) + endpoints.col(1)) / 2.);
95  const double tg_contribution =
96  tangential_components.col(edge_idx).transpose() *
97  eval_bdc_at_midpoint;
98 
99  auto gdof_indices{dofh_edge.GlobalDofIndices(*edge)};
100  LF_ASSERT_MSG(
101  gdof_indices.size() == 1,
102  "Edge should have 1 DOF for edge element basis functions");
103  // Set flags and values, NO accumulation!
104  for (const lf::assemble::gdof_idx_t gdof_idx : gdof_indices) {
105  ess_dof_select[gdof_idx] = {true, tg_contribution};
106  }
107  }
108  edge_idx++;
109  }
110  }
111  // Apply the boundary conditions to the linear system
112  lf::assemble::FixFlaggedSolutionCompAlt<double>(
113  [&ess_dof_select](lf::assemble::glb_idx_t dof_idx) {
114  return ess_dof_select[dof_idx];
115  },
116  A, phi);
117 }
