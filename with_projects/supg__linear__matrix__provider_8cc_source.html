6 double Diameter(const lf::mesh::Entity& entity) {
7  const lf::geometry::Geometry* geo_p = entity.Geometry();
8  Eigen::MatrixXd corners = lf::geometry::Corners(*geo_p);
9 
10  switch (entity.RefEl()) {
11  case lf::base::RefEl::kTria(): {
12  // Diameter of a triangle corresponds to the longest edge
13  Eigen::Vector2d e0 = corners.col(1) - corners.col(0);
14  Eigen::Vector2d e1 = corners.col(2) - corners.col(1);
15  Eigen::Vector2d e2 = corners.col(0) - corners.col(1);
16  return std::max(e0.norm(), std::max(e1.norm(), e2.norm()));
17  }
18  case lf::base::RefEl::kQuad(): {
19  // Diameter of a (convex) quadrilateral corresponds to the longer diagonal
20  Eigen::Vector2d d0 = corners.col(2) - corners.col(0);
21  Eigen::Vector2d d1 = corners.col(3) - corners.col(1);
22  return std::max(d0.norm(), d1.norm());
23  }
24  default: {
25  LF_ASSERT_MSG(false,
26  "Diameter not available for " << entity.RefEl().ToString());
27  return 0.0;
28  }
29  }
30 }

34  double const v_norm = v.lpNorm<Eigen::Infinity>();
35 
36  if (v_norm * h / (2 * eps) <= 1.0) {
37  return h * h / eps;
