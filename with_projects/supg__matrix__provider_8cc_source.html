10 void EnforceBoundaryConditions(
11  const std::shared_ptr<lf::uscalfe::UniformScalarFESpace<double>>& fe_space,
12  lf::assemble::COOMatrix<double>& A, Eigen::VectorXd& phi,
13  std::function<double(const Eigen::Matrix<double, 2, 1, 0>&)> dirichlet) {
14  lf::mesh::utils::MeshFunctionGlobal<decltype(dirichlet)> mf_g_dirichlet{
15  dirichlet};
16  lf::mesh::utils::AllCodimMeshDataSet<bool> bd_flags(fe_space->Mesh(), false);
17  // set a fixed epsilon value for the geometric test involving double
18  // comparison
19  const double kTol = 1e-8;
20  // Loop over all edges
21  for (const auto& edge : fe_space->Mesh()->Entities(1)) {
22  LF_ASSERT_MSG(edge->RefEl() == lf::base::RefEl::kSegment(),
23  "Entity should be an edge");
24  const lf::geometry::Geometry* geo_ptr = edge->Geometry();
25  const Eigen::MatrixXd corners = lf::geometry::Corners(*geo_ptr);
26  // Check if the edge lies on $\Gamma_{\mathrm{in}}$ (geometric test)
27  if ((corners(0, 0) + corners(0, 1)) / 2. < /*> 1. -*/ kTol ||
28  (corners(1, 0) + corners(1, 1)) / 2. < kTol) {
29  // Add the edge to the flagged entities
30  bd_flags(*edge) = true;
31  }
32  }
33  // Loop over all Points
34  for (const auto& point : fe_space->Mesh()->Entities(2)) {
35  LF_ASSERT_MSG(point->RefEl() == lf::base::RefEl::kPoint(),
36  "Entity should be an edge");
37  const lf::geometry::Geometry* geo_ptr = point->Geometry();
38  const Eigen::VectorXd coords = lf::geometry::Corners(*geo_ptr);
39  // Check if the node lies on $\Gamma_{\mathrm{in}}$ (geometric test)
40  if (coords(0) < /*> 1. -*/ kTol || coords(1) < kTol) {
41  // Add the edge to the flagged entities
42  bd_flags(*point) = true;
43  }
44  }
45  auto flag_values{lf::fe::InitEssentialConditionFromFunction(
46  *fe_space, bd_flags, mf_g_dirichlet)};
47 
48  lf::assemble::FixFlaggedSolutionCompAlt<double>(
49  [&flag_values](lf::assemble::glb_idx_t dof_idx) {
50  return flag_values[dof_idx];
51  },
52  A, phi);
53 }
