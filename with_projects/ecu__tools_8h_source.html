35 template <typename MF>
36 double EvaluateMeshFunction(std::shared_ptr<const lf::mesh::Mesh> mesh_p, MF mf,
37  const Eigen::Vector2d& global,
38  double tol = 10E-10) {
39  // Cell-wise search for the cell containing the point
40  for (const lf::mesh::Entity* e : mesh_p->Entities(0)) {
41  LF_ASSERT_MSG(lf::base::RefEl::kTria() == e->RefEl(),
42  "Only triangular cells are supported");
43  // Compute geometric information for the cell
44  const lf::geometry::Geometry* geo_ptr = e->Geometry();
45  Eigen::MatrixXd corners = lf::geometry::Corners(*geo_ptr);
46 
47  // transform global coordinates to local coordinates on the cell
48  Eigen::Matrix2d A;
49  A << corners.col(1) - corners.col(0), corners.col(2) - corners.col(0);
50  Eigen::Vector2d b;
51  b << global - corners.col(0);
52  Eigen::Vector2d local_coords = A.fullPivLu().solve(b);
53  // Check if the point is inside the reference triangle
54  if (local_coords(0) >= 0.0 - tol && local_coords(1) >= 0.0 - tol &&
55  local_coords(0) + local_coords(1) <= 1.0 + tol) {
56  return mf(*e, local_coords)[0];
57  }
58  return 0.0;
59  }
60 }

72 template <typename MF>
73 std::vector<double> EvaluateMeshFunction(
74  std::shared_ptr<const lf::mesh::Mesh> mesh_p, MF mf,
75  const std::vector<Eigen::Vector2d>& global, double tol = 10E-10) {
76  const unsigned int num_points = global.size();
77  std::vector<double> result(num_points);
78  std::vector<bool> computed(num_points, false);
79  unsigned int counter = 0;
80  for (const lf::mesh::Entity* e : mesh_p->Entities(0)) {
81  LF_ASSERT_MSG(lf::base::RefEl::kTria() == e->RefEl(),
82  "Only triangular cells are supported");
83  // Compute geometric information for the cell
84  const lf::geometry::Geometry* geo_ptr = e->Geometry();
85  Eigen::MatrixXd corners = lf::geometry::Corners(*geo_ptr);
86 
87  // transform global coordinates to local coordinates on the cell
88  Eigen::Matrix2d A;
89  A << corners.col(1) - corners.col(0), corners.col(2) - corners.col(0);
90  Eigen::FullPivLU<Eigen::Matrix2d> lu = A.fullPivLu();
91 
92  for (unsigned int i = 0; i < num_points; ++i) {
93  if (!computed[i]) {
94  // Compute local coordinates
95  Eigen::Vector2d b = global[i] - corners.col(0);
96  Eigen::Vector2d local_coords = lu.solve(b);
97 
98  // Check if the point is inside the reference triangle
99  if (local_coords(0) >= 0.0 - tol && local_coords(1) >= 0.0 - tol &&
100  local_coords(0) + local_coords(1) <= 1.0 + tol) {
101  // Evaluate the mesh function at the point
102  result[i] = mf(*e, local_coords)[0];
103  computed[i] = true;
104  counter++;
105 
106  if (counter == num_points) {
107  return result;
108  }
109  }
110  }
111  }
112  }
113  return result;
114 }
