80 
81  Eigen::Matrix<double, 3, 3> delta;
82  delta.block<3, 2>(0, 0) = corners.transpose();
83  delta.block<3, 1>(0, 2) = Eigen::Vector3d::Ones();
84  const double det = delta.determinant();
85  // rotation matrix of angle pi/2 is: 0, -1,
86  // 1, 0
87  const Eigen::Matrix<double, 2, 2> rotation_matrix =
88  (Eigen::Matrix<double, 2, 2>() << 0, -1, 1, 0).finished();
89  for (int i = 0; i < 3; ++i) {
90  const Eigen::Vector2d edge = corners.col((i + 1) % 3) - corners.col(i);
91  normals.col(i) = rotation_matrix * edge;
92  normals.col(i) /= normals.col(i).norm();
93  }
94  if (det > 0) {
