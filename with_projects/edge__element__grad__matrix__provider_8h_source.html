62 template <typename SCALAR, typename FUNCTOR>
63 Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic>
64 EdgeElementGradMatrixProvider<SCALAR, FUNCTOR>::Eval(
65  const lf::mesh::Entity& entity) const {
66  // Only triangles are supported
67  LF_VERIFY_MSG(entity.RefEl() == lf::base::RefEl::kTria(),
68  "Unsupported cell type" << entity.RefEl());
69 
70  // Get the geometry of the cell
71  const lf::geometry::Geometry* geo_ptr = entity.Geometry();
72  const Eigen::MatrixXd corners = lf::geometry::Corners(*geo_ptr);
73  const double area = lf::geometry::Volume(*geo_ptr);
74  LF_ASSERT_MSG(area > 0, "Area of cell must be positive");
75 
76  // Compute signed area of the cell for checking the orientation of vertices
77  const double signed_area =
78  (corners(0, 1) - corners(0, 0)) * (corners(1, 2) - corners(1, 0)) -
79  (corners(0, 2) - corners(0, 0)) * (corners(1, 1) - corners(1, 0));
80  // Set orientation based on signed area
81  const bool is_clockwise = (signed_area < 0);
82 
83  // Compute normals of vertices of the triangular cell
84  Eigen::Matrix3d X;
85  X.col(0) = Eigen::Vector3d::Ones();
86  X.rightCols(2) = corners.transpose();
87  const Eigen::MatrixXd temporary_gradients = X.inverse().bottomRows(2);
88  const Eigen::MatrixXd vertex_normals = -temporary_gradients;
89 
90  const size_t num_local_dofs = fe_space_->LocGlobMap().NumLocalDofs(entity);
91  LF_ASSERT_MSG(num_local_dofs == 6 || num_local_dofs == 3,
92  "Only quadratic or linear FE spaces are supported");
93  Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic> element_matrix(
94  num_local_dofs, num_local_dofs);
95  element_matrix.setZero();
96 
97  if (num_local_dofs == 3) {
98  // linear FE space
99  const lf::uscalfe::FeLagrangeO1Tria<SCALAR> hat_function;
100  // construct basis functions from curl of reference hat functions
101  const Eigen::MatrixXd reference_grads =
102  hat_function.GradientsReferenceShapeFunctions(Eigen::VectorXd::Zero(2))
103  .transpose();
104  // compute jacobian inverse gramian
105  const Eigen::MatrixXd j_inv_trans =
106  geo_ptr->JacobianInverseGramian(Eigen::VectorXd::Zero(2));
107  // compute gradients
108  const Eigen::MatrixXd grads = j_inv_trans * reference_grads;
109 
110  // edge element construction
111  Eigen::MatrixXd edge_element(grads.rows(), 3);
112  for (int i = 0; i < 3; ++i) {
113  edge_element.col(i) = grads.col((i + 1) % 3) - grads.col(i);
114  }
115  // correct orientation
116  auto edge_orientations = entity.RelativeOrientations();
117  for (int i = 0; i < 3; ++i) {
118  edge_element.col(i) *= lf::mesh::to_sign(edge_orientations[i]);
119  }
120  // compute part of the element matrix -- what is given by \int_T \nabla
121  // \lambda_i \psi_{e_j}
122  const Eigen::Matrix3d contribution_element_matrix =
123  (area / 3.0) * edge_element.transpose() * grads;
124 
125  // Row-vector of barycentric coordinate functions based on global
126  // coordinates of nodes
127  auto bary_functions =
128  [area, corners, is_clockwise](
129  const Eigen::Vector2d& xh) -> Eigen::Matrix<double, 1, 3> {
130  Eigen::Matrix<double, 1, 3> bary;
131  const double coeff = 1.0 / (2.0 * area);
132  // barycentric function 1
133  bary.col(0) = coeff * (xh - corners.col(1)).transpose() *
134  (Eigen::Vector2d(2, 1) << corners(1, 1) - corners(1, 2),
135  corners(0, 2) - corners(0, 1))
136  .finished();
137  // bary 2
138  bary.col(1) = coeff * (xh - corners.col(2)).transpose() *
139  (Eigen::Vector2d(2, 1) << corners(1, 2) - corners(1, 0),
140  corners(0, 0) - corners(0, 2))
141  .finished();
142  // bary 3
143  bary.col(2) = coeff * (xh - corners.col(0)).transpose() *
144  (Eigen::Vector2d(2, 1) << corners(1, 0) - corners(1, 1),
145  corners(0, 1) - corners(0, 0))
146  .finished();
147  if (is_clockwise) {
148  return -bary;
149  }
150  return bary;
151  };
152 
153  // Matrix of gradients of barycentric coordinate functions based on global
154  // coordinates of nodes
155  auto bary_functions_grad =
156  [area, corners, is_clockwise](
157  const Eigen::Vector2d& xh) -> Eigen::Matrix<double, 2, 3> {
158  Eigen::Matrix<double, 2, 3> bary_grad;
159  const double coeff = 1.0 / (2.0 * area);
160  // bary 1
161  bary_grad.col(0) =
162  coeff * (Eigen::Vector2d(2, 1) << corners(1, 1) - corners(1, 2),
163  corners(0, 2) - corners(0, 1))
164  .finished();
165  // bary 2
166  bary_grad.col(1) =
167  coeff * (Eigen::Vector2d(2, 1) << corners(1, 2) - corners(1, 0),
168  corners(0, 0) - corners(0, 2))
169  .finished();
170  // bary 3
171  bary_grad.col(2) =
172  coeff * (Eigen::Vector2d(2, 1) << corners(1, 0) - corners(1, 1),
173  corners(0, 1) - corners(0, 0))
174  .finished();
175  if (is_clockwise) {
176  return -bary_grad;
177  }
178  return bary_grad;
179  };
180  // compute psi_{e_i}(x_i)
181  auto edge_elem_basis = [edge_orientations, bary_functions,
182  bary_functions_grad](const Eigen::Vector2d& x) {
183  Eigen::Matrix<double, 2, 3> edge_elem_basis;
184  Eigen::Matrix<double, 1, 3> basis_lambda = bary_functions(x);
185  Eigen::Matrix<double, 2, 3> basis_lambda_grad = bary_functions_grad(x);
186  edge_elem_basis.col(0) = basis_lambda_grad.col(1) * basis_lambda.col(0) -
187  basis_lambda_grad.col(0) * basis_lambda.col(1);
188  edge_elem_basis.col(1) = basis_lambda_grad.col(2) * basis_lambda.col(1) -
189  basis_lambda_grad.col(1) * basis_lambda.col(2);
190  edge_elem_basis.col(2) = basis_lambda_grad.col(0) * basis_lambda.col(2) -
191  basis_lambda_grad.col(2) * basis_lambda.col(0);
192 
193  // correct orientation
194  edge_elem_basis.col(0) *= lf::mesh::to_sign(edge_orientations[0]);
195  edge_elem_basis.col(1) *= lf::mesh::to_sign(edge_orientations[1]);
196  edge_elem_basis.col(2) *= lf::mesh::to_sign(edge_orientations[2]);
197 
198  return edge_elem_basis;
199  };
200  // account for contribution that is row-wise constant -- (u \cdot
201  // psi_{e_i})(x_i) contribution is 0 iff the vertex x_i is not part of the
202  // upwind triangle
203  for (int i = 0; i < 3; ++i) {
204  Eigen::Vector2d velocity_at_vertex = v_(corners.col(i));
205  // compute contribution
206  double contribution =
207  velocity_at_vertex.transpose() *
208  edge_elem_basis(
209  (Eigen::Vector2d() << corners(0, i), corners(1, i)).finished())
210  .col(i);
211 
212  if (velocity_at_vertex.dot(vertex_normals.col((i + 2) % 3)) >= 0 &&
213  velocity_at_vertex.dot(vertex_normals.col((i + 1) % 3)) >= 0) {
214  // vertex is part of the upwind triangle
215  element_matrix += contribution * contribution_element_matrix;
216  } else {
217  // vertex is not part of the upwind triangle
218  element_matrix += Eigen::Matrix3d::Zero();
219  }
220  }
221 
222  return element_matrix;
223  } // end linear FE case
224  else {
225  // quadratic FE space
226  // todo
227  return element_matrix;
228  } // end quadratic FE case
229 }
