102 template <typename SCALAR, typename FUNCTOR>
103 Eigen::Matrix<SCALAR, 6, 6>
104 StableConvectionUpwindMatrixProvider<SCALAR, FUNCTOR>::Eval(
105  const lf::mesh::Entity &entity) {
106  const lf::geometry::Geometry *geo_ptr = entity.Geometry();
107  const Eigen::MatrixXd corners = lf::geometry::Corners(*geo_ptr);
108  const double area = lf::geometry::Volume(*geo_ptr);
109  LF_ASSERT_MSG(area > 0, "Area of cell must be positive");
110 
111  const size_t num_local_dofs = fe_space_->LocGlobMap().NumLocalDofs(entity);
112  LF_ASSERT_MSG(num_local_dofs == 6, "Only quadratic FE spaces are supported");
113  Eigen::Matrix<SCALAR, 6, 6> element_matrix =
114  Eigen::Matrix<SCALAR, 6, 6>::Zero();
115  // Compute normals of vertices of the triangular cell
116  Eigen::Matrix3d X;
117  X.col(0) = Eigen::Vector3d::Ones();
118  X.rightCols(2) = corners.transpose();
119  Eigen::MatrixXd temporary_gradients = X.inverse().bottomRows(2);
120  Eigen::MatrixXd vertex_normals = -temporary_gradients;
121 
122  // Compute signed area of the cell for checking the orientation of vertices
123  const double signed_area =
124  (corners(0, 1) - corners(0, 0)) * (corners(1, 2) - corners(1, 0)) -
125  (corners(0, 2) - corners(0, 0)) * (corners(1, 1) - corners(1, 0));
126  // Set orientation based on signed area
127  const bool is_clockwise = (signed_area < 0);
128 
129  // Compute edge midpoint coordinates - only needed for quadratic FE space
130  Eigen::MatrixXd midpoints(2, 3);
131  for (int i = 0; i < 3; ++i) {
132  midpoints.col(i) = 0.5 * (corners.col((i + 1) % 3) + corners.col(i));
133  }
134  // Compute barycenter of entity
135  Eigen::Vector2d barycenter = corners.rowwise().mean();
136  Eigen::Vector2d test_bary =
137  (corners.col(0) + corners.col(1) + corners.col(2)) / 3.0;
138  LF_ASSERT_MSG(barycenter == test_bary, "Barycenter computation failed");
139 
140  // Prepare matrices for the computation of the element matrix and separate
141  // implementation for linear and quadratic FE space
142  Eigen::MatrixXd velocities(2, num_local_dofs);
143  Eigen::MatrixXd all_nodes(2, num_local_dofs);
144  std::vector<double> local_masses;
145 
146  all_nodes << corners, midpoints;
147  velocities << v_(all_nodes.col(0)), v_(all_nodes.col(1)),
148  v_(all_nodes.col(2)), v_(all_nodes.col(3)), v_(all_nodes.col(4)),
149  v_(all_nodes.col(5));
150 
151  for (const lf::mesh::Entity *e : entity.SubEntities(2)) {
152  local_masses.push_back(masses_vertices_(*e));
153  }
154  for (const lf::mesh::Entity *e : entity.SubEntities(1)) {
155  local_masses.push_back(masses_edges_(*e));
156  }
157 
158  LF_ASSERT_MSG(local_masses.size() == 6,
159  "There must be six masses, one for each basis function");
160 
161  // Row-vector of barycentric coordinate functions based on global coordinates
162  // of nodes
163  auto bary_functions =
164  [area, corners,
165  is_clockwise](const Eigen::Vector2d &xh) -> Eigen::Matrix<double, 1, 3> {
166  Eigen::Matrix<double, 1, 3> bary;
167  const double coeff = 1.0 / (2.0 * area);
168  // barycentric function 1
169  bary.col(0) = coeff * (xh - corners.col(1)).transpose() *
170  (Eigen::Vector2d(2, 1) << corners(1, 1) - corners(1, 2),
171  corners(0, 2) - corners(0, 1))
172  .finished();
173  // bary 2
174  bary.col(1) = coeff * (xh - corners.col(2)).transpose() *
175  (Eigen::Vector2d(2, 1) << corners(1, 2) - corners(1, 0),
176  corners(0, 0) - corners(0, 2))
177  .finished();
178  // bary 3
179  bary.col(2) = coeff * (xh - corners.col(0)).transpose() *
180  (Eigen::Vector2d(2, 1) << corners(1, 0) - corners(1, 1),
181  corners(0, 1) - corners(0, 0))
182  .finished();
183  if (is_clockwise) {
184  return -bary;
185  }
186  return bary;
187  };
188 
189  // Matrix of gradients of barycentric coordinate functions based on global
190  // coordinates of nodes
191  auto bary_functions_grad =
192  [area, corners,
193  is_clockwise](const Eigen::Vector2d &xh) -> Eigen::Matrix<double, 2, 3> {
194  Eigen::Matrix<double, 2, 3> bary_grad;
195  const double coeff = 1.0 / (2.0 * area);
196  // bary 1
197  bary_grad.col(0) =
198  coeff * (Eigen::Vector2d(2, 1) << corners(1, 1) - corners(1, 2),
199  corners(0, 2) - corners(0, 1))
200  .finished();
201  // bary 2
202  bary_grad.col(1) =
203  coeff * (Eigen::Vector2d(2, 1) << corners(1, 2) - corners(1, 0),
204  corners(0, 0) - corners(0, 2))
205  .finished();
206  // bary 3
207  bary_grad.col(2) =
208  coeff * (Eigen::Vector2d(2, 1) << corners(1, 0) - corners(1, 1),
209  corners(0, 1) - corners(0, 0))
210  .finished();
211  if (is_clockwise) {
212  return -bary_grad;
213  }
214  return bary_grad;
215  };
216 
217  auto localShapeFunctions =
218  [bary_functions](
219  const Eigen::Vector2d xh) -> Eigen::Matrix<double, 1, 6> {
220  Eigen::Matrix<double, 1, 6> shapeFunctions;
221  Eigen::Matrix<double, 1, 3> temp = bary_functions(xh);
222  shapeFunctions << temp(0, 0) * (2 * temp(0, 0) - 1),
223  temp(0, 1) * (2 * temp(0, 1) - 1), temp(0, 2) * (2 * temp(0, 2) - 1),
224  4 * temp(0, 0) * temp(0, 1), 4 * temp(0, 1) * temp(0, 2),
225  4 * temp(0, 0) * temp(0, 2);
226  return shapeFunctions;
227  };
228 
229  // Matrix of gradients of local shape functions based on global coordinates of
230  // nodes in quadratic Lagrangian FE space
231  auto gradientsLocalShapeFunctions =
232  [bary_functions, bary_functions_grad](
233  const Eigen::Vector2d xh) -> Eigen::Matrix<double, 2, 6> {
234  Eigen::Matrix<double, 2, 6> gradients;
235  // barycentric coordinate functions
236  Eigen::Matrix<double, 1, 3> temp = bary_functions(xh);
237  // Eigen::Matrix<double, 1, 3> temp = bary_new(xh);
238  Eigen::Matrix<double, 2, 3> grads_bary = bary_functions_grad(xh);
239  Eigen::RowVector3d l;
240  l << temp(0, 0), temp(0, 1), temp(0, 2);
241 
242  gradients.col(0) = grads_bary.col(0) * (4 * l[0] - 1);
243  gradients.col(1) = grads_bary.col(1) * (4 * l[1] - 1);
244  gradients.col(2) = grads_bary.col(2) * (4 * l[2] - 1);
245  gradients.col(3) =
246  4 * (grads_bary.col(0) * l[1] + grads_bary.col(1) * l[0]);
247  gradients.col(4) =
248  4 * (grads_bary.col(1) * l[2] + grads_bary.col(2) * l[1]);
249  gradients.col(5) =
250  4 * (grads_bary.col(0) * l[2] + grads_bary.col(2) * l[0]);
251  return gradients;
252  };
253 
254  // Compute the element matrix
255  for (int l = 0; l < 6; ++l) {
256  Eigen::Vector2d vl = velocities.col(l);
257  // compute product of v(a^l) with gradients of basis function of a^l
258  Eigen::Matrix<double, 2, 6> grads =
259  gradientsLocalShapeFunctions(all_nodes.col(l));
260  Eigen::Matrix<double, 1, 6> contribution = vl.transpose() * grads;
261 
262  // Vertex a^l is upwind iff product of v(a^l) with both adjacent normals is
263  // positive
264  if (l < 3) {
265  // first 3 nodes are vertices of triangle
266  if (vl.dot(vertex_normals.col((l + 2) % 3)) >= 0 &&
267  vl.dot(vertex_normals.col((l + 1) % 3)) >= 0) {
268  // a^l is upwind
269  element_matrix.row(l) = local_masses[l] * contribution;
270  // LF_ASSERT_MSG(element_matrix(l, 0) == 0, "Element matrix entry
271  // must be zero");
272  } else {
273  // a^l is not upwind
274  element_matrix.row(l) = Eigen::Vector<SCALAR, 6>::Zero();
275  }
276  } else {
277  // last 3 nodes are edge midpoints of triangle
278  const Eigen::Matrix<double, 2, 3> outward_normals =
279  ecu_scheme::assemble::computeOutwardNormalsTria(entity);
280  // Midpoint m^l is upwind iff product of v(m^l) with corresponding outward
281  // normals is positive
282  if (vl.dot(outward_normals.col(l % 3)) >= 0) {
283  // m^l is upwind
284  if (vl.dot(outward_normals.col(l % 3)) == 0) {
285  element_matrix.row(l) = 0.5 * local_masses[l] * contribution;
286  } else {
287  element_matrix.row(l) = local_masses[l] * contribution;
288  }
289  } else {
290  // m^l is not upwind
291  element_matrix.row(l) = Eigen::Vector<SCALAR, 6>::Zero();
292  }
293  }
294 
295  // Account for barycentric contribution for each row of the element matrix
296  Eigen::Matrix<double, 1, 6> barycenter_row_contribution =
297  masses_cells_(entity) *
298  (v_(barycenter).transpose() * gradientsLocalShapeFunctions(barycenter))
299  .cwiseProduct(localShapeFunctions(barycenter));
300  element_matrix.row(l) += barycenter_row_contribution;
301  }
302 
303  return element_matrix;
304 }
