57  const auto kExactSolution = [eps_for_refinement](const Eigen::Vector2d &x) {
58  return x(0) * x(1) * x(1) -
59  x(1) * x(1) * std::exp(2.0 * (x(0) - 1.0) / eps_for_refinement) -
60  x(0) * std::exp(3.0 * (x(1) - 1.0) / eps_for_refinement) +
61  std::exp(2.0 * (x(0) - 1.0) / eps_for_refinement +
62  3.0 * (x(1) - 1.0) / eps_for_refinement);
63  };
64  // Gradient of exact solution u
65  const auto kGradExactSolution =
66  [eps_for_refinement](const Eigen::Vector2d &x) {
67  Eigen::Vector2d grad;
68  grad(0) = x(1) * x(1) -
69  x(1) * x(1) * 2.0 / eps_for_refinement *
70  std::exp(2.0 * (x(0) - 1.0) / eps_for_refinement) -
71  std::exp(3.0 * (x(1) - 1.0) / eps_for_refinement) +
72  2.0 / eps_for_refinement *
73  std::exp(2.0 * (x(0) - 1.0) / eps_for_refinement +
74  3.0 * (x(1) - 1.0) / eps_for_refinement);
75  grad(1) = 2 * x(0) * x(1) -
76  2 * x(1) * std::exp(2 * (x(0) - 1.0) / eps_for_refinement) -
77  x(0) * 3.0 / eps_for_refinement *
78  std::exp(3 * (x(1) - 1.0) / eps_for_refinement) +
79  3.0 / eps_for_refinement *
80  std::exp(2 * (x(0) - 1.0) / eps_for_refinement +
81  3 * (x(1) - 1.0) / eps_for_refinement);
82  return grad;
83  };
84  // Laplacian of the exact solution
85  const auto kLaplaceExactSolution =
86  [eps_for_refinement](const Eigen::Vector2d &x) {
87  double const uxx = -4.0 / (eps_for_refinement * eps_for_refinement) *
88  x(1) * x(1) *
89  std::exp(2 * (x(0) - 1.0) / eps_for_refinement) +
90  4.0 / (eps_for_refinement * eps_for_refinement) *
91  std::exp(2 * (x(0) - 1.0) / eps_for_refinement +
92  3 * (x(1) - 1.0) / eps_for_refinement);
93  double const uyy =
94  2.0 * x(0) - 2.0 * std::exp(2 * (x(0) - 1.0) / eps_for_refinement) -
95  x(0) * 9.0 / (eps_for_refinement * eps_for_refinement) *
96  std::exp(3 * (x(1) - 1.0) / eps_for_refinement) +
97  9.0 / (eps_for_refinement * eps_for_refinement) *
98  std::exp(2 * (x(0) - 1.0) / eps_for_refinement +
99  3 * (x(1) - 1.0) / eps_for_refinement);
100  return uxx + uyy;
