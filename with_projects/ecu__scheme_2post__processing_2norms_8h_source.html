47  const Eigen::VectorXd local_weights = quad_rule.Weights();

53  Eigen::VectorXd det{e->Geometry()->IntegrationElement(local_ref_coords)};
54  // Get function values

56  // Compute local quadrature of the squared norm

58  for (lf::base::size_type i = 0; i < num_local_qpts; ++i) {
59  local_squared_sum += det(i) * local_weights(i) * sq_f(values[i]);
60  }
61  cell_errors(*e) = std::sqrt(local_squared_sum);
62  squared_sum += local_squared_sum;
63  }
