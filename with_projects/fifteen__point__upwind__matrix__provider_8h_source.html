134 Eigen::Matrix<SCALAR, 6, 6>
135 FifteenPointUpwindMatrixProvider<SCALAR, FUNCTOR>::Eval(
136  const lf::mesh::Entity &entity) {
137  const lf::geometry::Geometry *geo_ptr = entity.Geometry();
138  const Eigen::MatrixXd corners = lf::geometry::Corners(*geo_ptr);
139  const double area = lf::geometry::Volume(*geo_ptr);
140  LF_ASSERT_MSG(area > 0, "Area of cell must be positive");
141 
142  const size_t num_local_dofs = fe_space_->LocGlobMap().NumLocalDofs(entity);
143  LF_ASSERT_MSG(num_local_dofs == 6, "Only quadratic FE spaces are supported");
144  Eigen::Matrix<SCALAR, 6, 6> element_matrix =
145  Eigen::Matrix<SCALAR, 6, 6>::Zero();
146  // Compute normals of vertices of the triangular cell
147  Eigen::Matrix3d X;
148  X.col(0) = Eigen::Vector3d::Ones();
149  X.rightCols(2) = corners.transpose();
150  Eigen::MatrixXd temporary_gradients = X.inverse().bottomRows(2);
151  Eigen::MatrixXd vertex_normals = -temporary_gradients;
152 
153  // Compute signed area of the cell for checking the orientation of vertices
154  const double signed_area =
155  (corners(0, 1) - corners(0, 0)) * (corners(1, 2) - corners(1, 0)) -
156  (corners(0, 2) - corners(0, 0)) * (corners(1, 1) - corners(1, 0));
157  // Set orientation based on signed area
158  const bool is_clockwise = (signed_area < 0);
159 
160  // Compute edge midpoint coordinates - only needed for quadratic FE space
161  Eigen::MatrixXd midpoints(2, 3);
162  for (int i = 0; i < 3; ++i) {
163  midpoints.col(i) = 0.5 * (corners.col((i + 1) % 3) + corners.col(i));
164  }
165  // Compute the fifteen quadrature points for exactness of degree p:
166  // THIS IS FOR THE REFERENCE TRIANGLE
167  // 14
168  // | \
169  // 12 13
170  // | \
171  // 9 - 10 - 11
172  // | \
173  // 5 - 6 - - 7 - 8
174  // | \
175  // 0 -- 1 -- 2 -- 3 -- 4
176  // They have the form (i/p, j/p) for i,j in {0,1,...,p} and i+j <= p
177  Eigen::MatrixXd quadrature_points(2, 15);
178  int temp_index = 0; // temporary index to insert points into matrix
179  const int kMaxDegreeOfExactness = 4;
180  for (int i = 0; i <= kMaxDegreeOfExactness; ++i) {
181  for (int j = 0; j <= kMaxDegreeOfExactness; ++j) {
182  if (i + j > kMaxDegreeOfExactness) {
183  continue;
184  }
185  quadrature_points.col(temp_index) =
186  (Eigen::Vector2d(2, 1) << ((double)i / (double)kMaxDegreeOfExactness),
187  ((double)j / (double)kMaxDegreeOfExactness))
188  .finished();
189  temp_index++;
190  }
191  }
192  LF_ASSERT_MSG(temp_index == 15, "There must be 15 quadrature points");
193  // In order to compare the obtained points we instantiate the precomputed
194  // Quadrature rule for the reference triangle
195  const std::vector<std::tuple<Eigen::Vector2d, double, bool>>
196  quad_rule_reference = prepareFifteenPointQuadRule();
197 
198  // Prepare matrices for the computation of the element matrix and separate
199  // implementation for linear and quadratic FE space
200  Eigen::MatrixXd velocities(2, num_local_dofs);
201  Eigen::MatrixXd all_nodes(2, num_local_dofs);
202  std::vector<double> local_masses;
203 
204  all_nodes << corners, midpoints;
205  velocities << v_(all_nodes.col(0)), v_(all_nodes.col(1)),
206  v_(all_nodes.col(2)), v_(all_nodes.col(3)), v_(all_nodes.col(4)),
207  v_(all_nodes.col(5));
208 
209  // Prepare a vector to store the 9 different quad points not available in
210  // S_2^0(M)
211 
212  std::vector<Eigen::Vector2d> truncated_transformed_quad_points;
213  // Transform the quadrature points from reference coords to physical
214  // coordinates
215  Eigen::MatrixXd transformed_quadrature_points =
216  geo_ptr->Global(quadrature_points);
217  // Remove the quadrature points that are associated with vertices and edge
218  // midpoints We are interested only in the other 9 quad points
219  Eigen::MatrixXd edge_off_points(2, 6);
220  edge_off_points << transformed_quadrature_points.col(1),
221  transformed_quadrature_points.col(3),
222  transformed_quadrature_points.col(5),
223  transformed_quadrature_points.col(8),
224  transformed_quadrature_points.col(12),
225  transformed_quadrature_points.col(13);
226  Eigen::MatrixXd cell_center_points(2, 3);
227  cell_center_points << transformed_quadrature_points.col(6),
228  transformed_quadrature_points.col(7),
229  transformed_quadrature_points.col(10);
230 
231  for (const lf::mesh::Entity *e : entity.SubEntities(2)) {
232  local_masses.push_back(masses_vertices_(*e));
233  }
234  for (const lf::mesh::Entity *e : entity.SubEntities(1)) {
235  local_masses.push_back(masses_edge_midpoints_(*e));
236  }
237 
238  LF_ASSERT_MSG(local_masses.size() == 6,
239  "There must be six masses, one for each basis function");
240 
241  // Row-vector of barycentric coordinate functions based on global coordinates
242  // of nodes
243  auto bary_functions =
244  [area, corners,
245  is_clockwise](const Eigen::Vector2d &xh) -> Eigen::Matrix<double, 1, 3> {
246  Eigen::Matrix<double, 1, 3> bary;
247  const double coeff = 1.0 / (2.0 * area);
248  // barycentric function 1
249  bary.col(0) = coeff * (xh - corners.col(1)).transpose() *
250  (Eigen::Vector2d(2, 1) << corners(1, 1) - corners(1, 2),
251  corners(0, 2) - corners(0, 1))
252  .finished();
253  // bary 2
254  bary.col(1) = coeff * (xh - corners.col(2)).transpose() *
255  (Eigen::Vector2d(2, 1) << corners(1, 2) - corners(1, 0),
256  corners(0, 0) - corners(0, 2))
257  .finished();
258  // bary 3
259  bary.col(2) = coeff * (xh - corners.col(0)).transpose() *
260  (Eigen::Vector2d(2, 1) << corners(1, 0) - corners(1, 1),
261  corners(0, 1) - corners(0, 0))
262  .finished();
263  if (is_clockwise) {
264  return -bary;
265  }
266  return bary;
267  };
268 
269  // Matrix of gradients of barycentric coordinate functions based on global
270  // coordinates of nodes
271  auto bary_functions_grad =
272  [area, corners,
273  is_clockwise](const Eigen::Vector2d &xh) -> Eigen::Matrix<double, 2, 3> {
274  Eigen::Matrix<double, 2, 3> bary_grad;
275  const double coeff = 1.0 / (2.0 * area);
276  // bary 1
277  bary_grad.col(0) =
278  coeff * (Eigen::Vector2d(2, 1) << corners(1, 1) - corners(1, 2),
279  corners(0, 2) - corners(0, 1))
280  .finished();
281  // bary 2
282  bary_grad.col(1) =
283  coeff * (Eigen::Vector2d(2, 1) << corners(1, 2) - corners(1, 0),
284  corners(0, 0) - corners(0, 2))
285  .finished();
286  // bary 3
287  bary_grad.col(2) =
288  coeff * (Eigen::Vector2d(2, 1) << corners(1, 0) - corners(1, 1),
289  corners(0, 1) - corners(0, 0))
290  .finished();
291  if (is_clockwise) {
292  return -bary_grad;
293  }
294  return bary_grad;
295  };

297  auto localShapeFunctions =
298  [bary_functions](
299  const Eigen::Vector2d xh) -> Eigen::Matrix<double, 1, 6> {
300  Eigen::Matrix<double, 1, 6> shapeFunctions;
301  Eigen::Matrix<double, 1, 3> temp = bary_functions(xh);
302  shapeFunctions << temp(0, 0) * (2 * temp(0, 0) - 1),
303  temp(0, 1) * (2 * temp(0, 1) - 1), temp(0, 2) * (2 * temp(0, 2) - 1),
304  4 * temp(0, 0) * temp(0, 1), 4 * temp(0, 1) * temp(0, 2),
305  4 * temp(0, 0) * temp(0, 2);
306  return shapeFunctions;
307  };
308 
309  // Matrix of gradients of local shape functions based on global coordinates of
310  // nodes in quadratic Lagrangian FE space
311  auto gradientsLocalShapeFunctions =
312  [bary_functions, bary_functions_grad](
313  const Eigen::Vector2d xh) -> Eigen::Matrix<double, 2, 6> {
314  Eigen::Matrix<double, 2, 6> gradients;
315  // barycentric coordinate functions
316  Eigen::Matrix<double, 1, 3> temp = bary_functions(xh);
317  // Eigen::Matrix<double, 1, 3> temp = bary_new(xh);
318  Eigen::Matrix<double, 2, 3> grads_bary = bary_functions_grad(xh);
319  Eigen::RowVector3d l;
320  l << temp(0, 0), temp(0, 1), temp(0, 2);
321 
322  gradients.col(0) = grads_bary.col(0) * (4 * l[0] - 1);
323  gradients.col(1) = grads_bary.col(1) * (4 * l[1] - 1);
324  gradients.col(2) = grads_bary.col(2) * (4 * l[2] - 1);
325  gradients.col(3) =
326  4 * (grads_bary.col(0) * l[1] + grads_bary.col(1) * l[0]);
327  gradients.col(4) =
328  4 * (grads_bary.col(1) * l[2] + grads_bary.col(2) * l[1]);
329  gradients.col(5) =
330  4 * (grads_bary.col(0) * l[2] + grads_bary.col(2) * l[0]);
331  return gradients;
332  };

334  // Contribution for edge quad points that are not midpoints of edges
335  std::vector<double> local_masses_edge_off;
336  for (const lf::mesh::Entity *e : entity.SubEntities(1)) {
337  local_masses_edge_off.push_back(masses_edges_(*e));
338  }
339  // Compute the element matrix
340  for (int l = 0; l < 6; ++l) {
341  Eigen::Vector2d vl = velocities.col(l);
342  // compute product of v(a^l) with gradients of basis function of a^l
343  Eigen::Matrix<double, 2, 6> grads =
344  gradientsLocalShapeFunctions(all_nodes.col(l));
345  Eigen::Matrix<double, 1, 6> contribution = vl.transpose() * grads;
346 
347  // Vertex a^l is upwind iff product of v(a^l) with both adjacent normals is
348  // positive
349  if (l < 3) {
350  // first 3 nodes are vertices of triangle
351  if (vl.dot(vertex_normals.col((l + 2) % 3)) >= 0 &&
352  vl.dot(vertex_normals.col((l + 1) % 3)) >= 0) {
353  // a^l is upwind
354  element_matrix.row(l) = local_masses[l] * contribution;
355  LF_ASSERT_MSG(element_matrix(l, 0) == 0,
356  "Element matrix entry must be zero");
357  } else {
358  // a^l is not upwind
359  element_matrix.row(l) = Eigen::Vector<SCALAR, 6>::Zero();
360  }
361  } else {
362  // last 3 nodes are edge midpoints of triangle
363  const Eigen::Matrix<double, 2, 3> outward_normals =
364  ecu_scheme::assemble::computeOutwardNormalsTria(entity);
365  // Midpoint m^l is upwind iff product of v(m^l) with corresponding outward
366  // normals is positive
367  if (vl.dot(outward_normals.col(l % 3)) >= 0) {
368  // m^l is upwind
369  element_matrix.row(l) = local_masses[l] * contribution;
370  } else {
371  // m^l is not upwind
372  element_matrix.row(l) = Eigen::Vector<SCALAR, 6>::Zero();
373  }
374  }
375 
376  // Account for contribution of the other 15 - 6 = 9 quadrature points.
377  // Contribution is added to each row of the element matrix.
378 
379  // Make a case distinction since LehrFEM++ edge numbering is a bit different
380  // than the numbering from how the quad points are created
381  int temp_local_index_edges = 0;
382  for (auto off_edge : edge_off_points.colwise()) {
383  Eigen::Matrix<double, 1, 6> off_edge_contribution =
384  (v_(off_edge).transpose() * gradientsLocalShapeFunctions(off_edge))
385  .cwiseProduct(localShapeFunctions(off_edge));
386  // add upwind property because of edges
387  const Eigen::Matrix<double, 2, 3> outward_normals_off =
388  ecu_scheme::assemble::computeOutwardNormalsTria(entity);
389  // if(vl.dot(outward_normals_off.col(l%3)) < 0){
390  // // edge not upwind
391  // off_edge_contribution = Eigen::Matrix<double, 1, 6>::Zero();
392  // }else{
393  if (temp_local_index_edges == 0 || temp_local_index_edges == 1) {
394  off_edge_contribution *= local_masses_edge_off[0];
395  } else if (temp_local_index_edges == 2 || temp_local_index_edges == 4) {
396  off_edge_contribution *= local_masses_edge_off[1];
397  } else if (temp_local_index_edges == 3 || temp_local_index_edges == 5) {
398  off_edge_contribution *= local_masses_edge_off[2];
399  } else {
400  std::cerr << "There should be a total of 6 edge points that are not "
401  "edge midpoints"
402  << std::endl;
403  }
404  // }
405  element_matrix.row(l) += off_edge_contribution;
406  temp_local_index_edges++;
407  }
408  for (auto cell_node : cell_center_points.colwise()) {
409  Eigen::Matrix<double, 1, 6> cell_center_contribution =
410  masses_cells_(entity) *
411  (v_(cell_node).transpose() * gradientsLocalShapeFunctions(cell_node))
412  .cwiseProduct(localShapeFunctions(cell_node));
413  element_matrix.row(l) += cell_center_contribution;
414  }
415  }
416  return element_matrix;
417 }
