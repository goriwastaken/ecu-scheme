69 template <class MESHFUNCTION_V>
70 SUPGElementMatrixProvider<MESHFUNCTION_V>::SUPGElementMatrixProvider(
71  MESHFUNCTION_V &velocity, bool use_delta)
72  : velocity_(velocity), use_delta_(use_delta) {
73  const lf::uscalfe::FeLagrangeO2Tria<double> ref_fe_space;
74  LF_ASSERT_MSG(ref_fe_space.RefEl() == lf::base::RefEl::kTria(),
75  "This class only works for triangular elements");
76  LF_ASSERT_MSG(ref_fe_space.NumRefShapeFunctions() == 6,
77  "Quadratic Lagrange FE must have 6 local shape functions");
78  LF_ASSERT_MSG(qr_.RefEl() == lf::base::RefEl::kTria(),
79  "Quadrature rule only for triangular cells");
80  LF_ASSERT_MSG(qr_.NumPoints() == 6, "Quadrature rule must have 6 points");
81  // Obtain values and gradients of reference shape functions at
82  // quadrature nodes, see \lref{par:lfppparfe}
83  val_ref_lsf_ = ref_fe_space.EvalReferenceShapeFunctions(qr_.Points());
84  grad_ref_lsf_ = ref_fe_space.GradientsReferenceShapeFunctions(qr_.Points());
85 }

87 template <class MESHFUNCTION_V>
88 Eigen::Matrix<double, Eigen::Dynamic, Eigen::Dynamic>
89 SUPGElementMatrixProvider<MESHFUNCTION_V>::Eval(const lf::mesh::Entity &cell) {
90  // For quadratic Lagrange FE we have exactly 6 local shape functions
91  // Element matrix is a 6x6 matrix
92  Eigen::Matrix<double, 6, 6> elem_mat;
93  elem_mat.setZero();
94 
95  LF_ASSERT_MSG(cell.RefEl() == lf::base::RefEl::kTria(),
96  "Only implemented for triangles");
97  // Obtain geometry information
98  const lf::geometry::Geometry *geo_ptr = cell.Geometry();
99  LF_ASSERT_MSG(geo_ptr->DimGlobal() == 2,
100  "Only implemented for planar triangles");
101  // Gram determinant at quadrature points
102  const Eigen::VectorXd dets(geo_ptr->IntegrationElement(qr_.Points()));
103  LF_ASSERT_MSG(dets.size() == qr_.NumPoints(),
104  "Mismatch " << dets.size() << " <-> " << qr_.NumPoints());
105  // Fetch the transformation matrices for the gradients
106  const Eigen::MatrixXd JinvT(geo_ptr->JacobianInverseGramian(qr_.Points()));
107  LF_ASSERT_MSG(JinvT.cols() == 2 * qr_.NumPoints(),
108  "Mismatch " << JinvT.cols() << " <-> " << 2 * qr_.NumPoints());
109  // Obtain values of velocity at the 6 quadrature points
110  std::vector<Eigen::Vector2d> v_vals = velocity_(cell, qr_.Points());
111 
112  // two ways to compute delta coefficient, we use delta instead of delta^(1/2)
113  const Eigen::MatrixXd vertices{lf::geometry::Corners(*geo_ptr)};
114  // size of triangle
115  const double hK = std::max({(vertices.col(0) - vertices.col(1)).norm(),
116  (vertices.col(1) - vertices.col(2)).norm(),
117  (vertices.col(2) - vertices.col(0)).norm()});
118  // max modulus of velocity in quad nodes
119  const double velocity_max =
120  std::max_element(v_vals.begin(), v_vals.end(),
121  [](const Eigen::Vector2d &a, const Eigen::Vector2d &b) {
122  return a.norm() < b.norm();
123  })
124  ->norm();
125  const double kDelta = use_delta_ ? hK / (2.0 * velocity_max) : 1.0;
126  // End compute the delta coefficient
127 
128  for (int l = 0; l < 6; ++l) {
129  // Metric factor $\cob{|\det\Derv\Phibf_K(\wh{\zetabf}_{\ell})}$ scaled with
130  // quadrature weight $\cob{\omega_{\ell}}$
131  const double kFactor = qr_.Weights()[l] * dets[l];
132  // Compute transformed gradients $\cob{\Vt_{\ell}^i}$ of all local shape
133  // functions and collect them in the columns of a 2 x 6-matrix
134  const Eigen::Matrix<double, 2, 6> kTrfGrad =
135  JinvT.block(0, 2 * l, 2, 2) *
136  (grad_ref_lsf_.block(0, 2 * l, 6, 2).transpose());
137  // Compute the inner products of the transformed gradients with the
138  // velocity vector at the current quadrature point.
139  const Eigen::Matrix<double, 1, 6> mvec = v_vals[l].transpose() * kTrfGrad;
140  // Advective part of the SUPG element matrix
141  for (int j = 0; j < 6; ++j) {
142  for (int i = 0; i < 6; ++i) {
143  elem_mat(i, j) += kFactor * (kDelta * mvec[i] * mvec[j] +
144  mvec[j] * val_ref_lsf_(i, l));
145  }
146  }
147  }
148  return elem_mat;
149 }

152 template <typename VELOCITY>
153 lf::mesh::utils::CodimMeshDataSet<bool> flagNodesOnInflowBoundary(
154  const std::shared_ptr<const lf::mesh::Mesh> &mesh_p, VELOCITY velo) {
155  static_assert(lf::mesh::utils::isMeshFunction<VELOCITY>);
156  // Array for flags
157  lf::mesh::utils::CodimMeshDataSet<bool> nd_inflow_flags(mesh_p, 2, false);
158  // Reference coordinates of center of gravity of a triangle
159  const Eigen::MatrixXd c_hat = Eigen::Vector2d(1.0 / 3.0, 1.0 / 3.0);
160  // Reference coordinates of midpoints of edges
161  const Eigen::MatrixXd mp_hat =
162  (Eigen::Matrix<double, 2, 3>() << 0.5, 0.5, 0.0, 0.0, 0.5, 0.5)
163  .finished();
164  // Find edges (codim = 1) on the boundary
165  lf::mesh::utils::CodimMeshDataSet<bool> ed_bd_flags(
166  lf::mesh::utils::flagEntitiesOnBoundary(mesh_p, 1));
167  // Run through all cells of the mesh and determine
168  for (const lf::mesh::Entity *cell : mesh_p->Entities(0)) {
169  // Fetch geometry object for current cell
170  const lf::geometry::Geometry &K_geo{*(cell->Geometry())};
171  LF_ASSERT_MSG(cell->RefEl() == lf::base::RefEl::kTria(),
172  "Only implemented for triangles");
173  LF_ASSERT_MSG(K_geo.DimGlobal() == 2, "Mesh must be planar");
174  // Obtain physical coordinates of barycenter of triangle
175  const Eigen::Vector2d center{K_geo.Global(c_hat).col(0)};
176  // Get velocity values in the midpoints of the edges
177  auto velo_mp_vals = velo(*cell, mp_hat);
178  // Retrieve pointers to all edges of the triangle
179  nonstd::span<const lf::mesh::Entity *const> edges{cell->SubEntities(1)};
180  LF_ASSERT_MSG(edges.size() == 3, "Triangle must have three edges!");
181  for (int k = 0; k < 3; ++k) {
182  if (ed_bd_flags(*edges[k])) {
183  const lf::geometry::Geometry &ed_geo{*(edges[k]->Geometry())};
184  const Eigen::MatrixXd ed_pts{lf::geometry::Corners(ed_geo)};
185  // Direction vector of the edge
186  const Eigen::Vector2d dir = ed_pts.col(1) - ed_pts.col(0);
187  // Rotate counterclockwise by 90 degrees
188  const Eigen::Vector2d ed_normal = Eigen::Vector2d(dir(1), -dir(0));
189  // For adjusting direction of normal so that it points into the exterior
190  // of the domain
191  const int ori = (ed_normal.dot(center - ed_pts.col(0)) > 0) ? -1 : 1;
192  // Check angle of exterior normal and velocity vector
193  const int v_rel_ori =
194  ((velo_mp_vals[k].dot(ed_normal) > 0) ? 1 : -1) * ori;
195  if (v_rel_ori < 0) {
196  // Inflow: obtain endpoints of the edge and mark them
197  nonstd::span<const lf::mesh::Entity *const> endpoints{
198  edges[k]->SubEntities(1)};
199  LF_ASSERT_MSG(endpoints.size() == 2, "Edge must have two endpoints!");
200  nd_inflow_flags(*endpoints[0]) = true;
201  nd_inflow_flags(*endpoints[1]) = true;
202  }
203  }
204  }
205  }
206  return nd_inflow_flags;
207 }

216  lf::mesh::utils::MeshFunctionGlobal mf_eps{eps};
217  lf::mesh::utils::MeshFunctionGlobal mf_v{v};

236 
237  // RHS vector

240  lf::fe::ScalarLoadElementVectorProvider<double, decltype(mf_f)>
241  load_element_vector_provider(fe_space, mf_f);
242  lf::assemble::AssembleVectorLocally(0, dofh, load_element_vector_provider,
243  phi);
244

246  // Eigen::VectorXd g_coeffs = lf::fe::NodalProjection(*fe_space, mf_g);

254  // g_coeffs[dof_idx]};
255  // }, A, phi);
256  if (!flagInflow) {
257  EnforceBoundaryConditions(fe_space, A, phi, g);
258  } else {
259  EnforceBoundaryConditionsOnRotInflow(fe_space, A, phi, g);
260  }

265  // ess_bdc_flags_values{lf::fe::InitEssentialConditionFromFunction(*fe_space,

268  // return ess_bdc_flags_values[gdof_idx];
269  // }, A, phi);
270 
271  // SOLVE LINEAR SYSTEM
272  Eigen::SparseMatrix<double> A_crs = A.makeSparse();
273  Eigen::SparseLU<Eigen::SparseMatrix<double>> solver;
274  solver.compute(A_crs);
275  if (solver.info() != Eigen::Success) {
276  std::cerr << "LU decomposition failed for manufactured solution"

279  Eigen::VectorXd solution_vector = solver.solve(phi);
280  if (solver.info() != Eigen::Success) {

282  }
