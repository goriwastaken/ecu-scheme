37 Eigen::Vector2d linearFit(const Eigen::VectorXd& x, const Eigen::VectorXd& y) {
38  assert(x.rows() == y.rows());
39  Eigen::Matrix<double, Eigen::Dynamic, 2> X(x.rows(), 2);
40 
41  X.col(0) = Eigen::VectorXd::Constant(x.rows(), 1.0);
42  X.col(1) = x;
43 
44  return X.fullPivHouseholderQr().solve(y);
45 }

47 double eoc(Eigen::VectorXd& N, Eigen::VectorXd& err,
48  unsigned int from_index = 0) {
49  const unsigned dim = N.size();
50 
51  // truncate pre-asymptotic behaviour if desired
52  const unsigned int newdim = dim - from_index;
53  // compute log(N) and log(err) component-wise
54  auto logfun = [](double d) { return std::log(d); };
55  Eigen::VectorXd Nlog(newdim), errlog(newdim);
56  std::transform(N.data() + from_index, N.data() + dim, Nlog.data(), logfun);
57  std::transform(err.data() + from_index, err.data() + dim, errlog.data(),
58  logfun);
59 
60  Eigen::Vector2d polyfit = linearFit(Nlog, errlog);
61  double alpha = -polyfit[1];
62 
63  return alpha;
64 }

112 template <typename SCALAR, typename MF>
113 void convergence_report_oneform(
114  const ExperimentSolutionWrapper<SCALAR>& solution_collection_wrapper,
115  MF& mf_exact_solution, std::string experiment_name, bool isLinear = false) {
116  // generate location for output
117  std::string main_dir = PROJECT_BUILD_DIR "/results";
118  std::filesystem::create_directory(main_dir); // if necessary
119 
120  // // set Eigen csv format if needed
121  // const Eigen::IOFormat CSVFormat(Eigen::FullPrecision,
122  // Eigen::DontAlignCols, ", ", "\n");
123 
124  // create a csv file to store the results for plotting
125  std::ofstream L2norm_csv_file;
126  L2norm_csv_file.open(concat(main_dir, "/", experiment_name, "_L2error.csv"));
127  // include metadata about the experiment such as number of refinement levels
128  // and the diffusion coefficient used
129  L2norm_csv_file << "Number of refinement levels: "
130  << solution_collection_wrapper.refinement_levels << "\n";
131  L2norm_csv_file << "Diffusion coefficient: "
132  << solution_collection_wrapper.eps << "\n";
133  L2norm_csv_file << "No. of dofs,Meshwidth,Extrusion-Contraction Upwind"
134  << "\n";
135 
136  // define square functions for norms
137  auto square_scalar = [](SCALAR x) -> double {
138  return std::abs(x) * std::abs(x);
139  };
140  auto square_vector =
141  [](Eigen::Matrix<SCALAR, Eigen::Dynamic, 1> x) -> double {
142  return x.squaredNorm();
143  };
144 
145  // solution_collection_wrapper contains the mesh hierarchy used and hence each
146  // mesh corresponding to a refinement level
147  lf::refinement::MeshHierarchy& multi_mesh{
148  *solution_collection_wrapper.mesh_hierarchy_p};
149 
150  // get number of levels
151  auto L = multi_mesh.NumLevels();
152 
153  Eigen::VectorXd Ndof_array(L);
154  Eigen::VectorXd errors_array(L);
155  // perform computations on all levels
156  for (int l = 0; l < L; ++l) {
157  // get mesh for refinement level l
158  std::shared_ptr<const lf::mesh::Mesh> mesh_l{multi_mesh.getMesh(l)};
159 
160  // generate DOFHandler corresponding to edge element basis functions to
161  // report number of DOFs
162  const lf::assemble::DofHandler& dofh_edge =
163  isLinear ? lf::assemble::UniformFEDofHandler(
164  mesh_l, {{lf::base::RefEl::kSegment(), 1}})
165  : lf::assemble::UniformFEDofHandler(
166  mesh_l, {{lf::base::RefEl::kSegment(), 2},
167  {lf::base::RefEl::kTria(), 2}});
168  // compute meshwidth
169  const double kHMax =
170  ecu_scheme::post_processing::ComputeMeshWidthTria(mesh_l);
171  // get finite element space for refinement level l
172  std::shared_ptr<lf::uscalfe::UniformScalarFESpace<SCALAR>> fe_space;
173  if (isLinear) {
174  // Linear FE space
175  fe_space =
176  std::make_shared<lf::uscalfe::FeSpaceLagrangeO1<SCALAR>>(mesh_l);
177  } else {
178  // Quadratic FE space
179  fe_space =
180  std::make_shared<lf::uscalfe::FeSpaceLagrangeO2<SCALAR>>(mesh_l);
181  }
182  // get solution vector computed at refinement l
183  Eigen::VectorXd solution_vec =
184  solution_collection_wrapper.final_time_solutions.at(l);
185 
186  // Take finite element solution and wrap it into specialized mesh function
187  // for 1-forms
188  ecu_scheme::assemble::MeshFunctionOneForm<double> mf_fe_one_form(
189  solution_vec, mesh_l);
190  // Compute difference with exact solution (care that the
191  // lf::uscalfe::operator-() is used
192  auto mf_diff = mf_fe_one_form - mf_exact_solution;
193 
194  // Prepare computation of L2-error, generate a quadrature rule
195  lf::quad::QuadRule quad_rule =
196  lf::quad::make_QuadRule(lf::base::RefEl::kTria(), 10);
197  // Compute L2 error
198  const std::pair<double, lf::mesh::utils::CodimMeshDataSet<double>>
199  L2_error_bundle = ecu_scheme::post_processing::L2norm(
200  mesh_l, mf_diff, square_vector, quad_rule);
201 
202  Ndof_array(l) = dofh_edge.NumDofs();
203  errors_array(l) = std::get<0>(L2_error_bundle);
204 
205  // debug only - plot each mesh with matlab
206  // lf::io::writeMatlab(*(fe_space->Mesh()), concat(main_dir, "/",
207  // experiment_name, "_matlabmesh", std::to_string(l), ".m"));
208  // how to plot using lehrfem plot_
209 
210  // Add results to L2norm_csv_file file
211  L2norm_csv_file << dofh_edge.NumDofs() << "," << kHMax << ","
212  << std::get<0>(L2_error_bundle) << "\n";
213  std::cout << std::left << std::setw(10) << fe_space->LocGlobMap().NumDofs()
214  << std::left << std::setw(16) << dofh_edge.NumDofs() << std::left
215  << std::setw(16) << std::get<0>(L2_error_bundle)
216  << std::endl; // debug purpose
217  }
218  L2norm_csv_file.close();
219 
220  // Plot the computed L2error rate
221  double eoc_value = eoc(Ndof_array, errors_array);
222  std::cout << "EOC value: " << eoc_value << "\n";
223 }
