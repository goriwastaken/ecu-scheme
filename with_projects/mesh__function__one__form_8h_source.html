25 template <typename SCALAR>
26 std::vector<Eigen::Matrix<SCALAR, Eigen::Dynamic, 1>>
27 MeshFunctionOneForm<SCALAR>::operator()(
28  const lf::mesh::Entity& entity, const Eigen::MatrixXd& local_coords) const {
29  LF_ASSERT_MSG(entity.RefEl() == lf::base::RefEl::kTria(),
30  "Only implemented for triangular cells");
31 
32  std::vector<Eigen::Matrix<SCALAR, Eigen::Dynamic, 1>> result(
33  local_coords.cols());
34 
35  // Get geometric information about the cell
36  const lf::geometry::Geometry* geo_ptr = entity.Geometry();
37  Eigen::MatrixXd corners = lf::geometry::Corners(*geo_ptr);
38 
39  // Get basis basis functions
40  const lf::uscalfe::FeLagrangeO1Tria<SCALAR> fe_space;
41  const Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic> ref_grads =
42  fe_space.GradientsReferenceShapeFunctions(Eigen::VectorXd::Zero(2))
43  .transpose();
44  const Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic> jacobian_inv =
45  geo_ptr->JacobianInverseGramian(Eigen::VectorXd::Zero(2));
46 
47  // Compute gradients
48  const Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic> grads =
49  jacobian_inv * ref_grads;
50 
51  // Obtain reference shape functions
52  const Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic> lambda =
53  fe_space.EvalReferenceShapeFunctions(local_coords);
54 
55  // Take into account orientation
56  auto edgeOrientations = entity.RelativeOrientations();
57 
58  // Obtain global coordinates
59  std::vector<lf::base::size_type> global_indices(
60  fe_space.NumRefShapeFunctions());
61  auto edges = entity.SubEntities(1);
62  for (lf::base::size_type iter = 0; iter < 3; ++iter) {
63  global_indices[iter] = mesh_p_->Index(*edges[iter]);
64  }
65 
66  // Compute the one form
67  const int kShape = fe_space.NumRefShapeFunctions();
68  for (int i = 0; i < local_coords.cols(); ++i) {
69  Eigen::Matrix<SCALAR, Eigen::Dynamic, Eigen::Dynamic> basis_func(
70  grads.rows(), kShape);
71  for (int j = 0; j < kShape; ++j) {
72  basis_func.col(j) = lambda(j, i) * grads.col((j + 1) % kShape) -
73  lambda((j + 1) % kShape, i) * grads.col(j);
74  // account for orientation
75  basis_func.col(j) *= lf::mesh::to_sign(edgeOrientations[j]);
76  }
77 
78  result[i] = mu_(global_indices[0]) * basis_func.col(0) +
79  mu_(global_indices[1]) * basis_func.col(1) +
80  mu_(global_indices[2]) * basis_func.col(2);
81  }
82  return result;
83 }
