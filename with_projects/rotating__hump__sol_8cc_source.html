13  // set a fixed epsilon value for the geometric test involving double
14  // comparison
15  const double kTol = 1e-8;
16  // Loop over all edges
17  for (const auto& edge : fe_space->Mesh()->Entities(1)) {
18  LF_ASSERT_MSG(edge->RefEl() == lf::base::RefEl::kSegment(),
19  "Entity should be an edge");
20  const lf::geometry::Geometry* geo_ptr = edge->Geometry();
21  const Eigen::MatrixXd corners = lf::geometry::Corners(*geo_ptr);
22  // Check if the edge lies on $\Gamma_{\mathrm{in}}$ (geometric test)
23  if ((corners(0, 0) + corners(0, 1)) / 2. > 1. - kTol ||
24  (corners(1, 0) + corners(1, 1)) / 2. < kTol) {
25  // Add the edge to the flagged entities
26  bd_flags(*edge) = true;
27  }
28  }
29  // Loop over all Points
30  for (const auto& point : fe_space->Mesh()->Entities(2)) {
31  LF_ASSERT_MSG(point->RefEl() == lf::base::RefEl::kPoint(),
32  "Entity should be an edge");
33  const lf::geometry::Geometry* geo_ptr = point->Geometry();
34  const Eigen::VectorXd coords = lf::geometry::Corners(*geo_ptr);
35  // Check if the node lies on $\Gamma_{\mathrm{in}}$ (geometric test)
36  if (coords(0) > 1. - kTol || coords(1) < kTol) {
37  // Add the edge to the flagged entities
38  bd_flags(*point) = true;
39  }
40  }

42  *fe_space, bd_flags, mf_g_dirichlet)};
