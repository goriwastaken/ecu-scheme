80  auto mesh_p = fe_space_->Mesh();
81  const lf::assemble::DofHandler &dofh{fe_space_->LocGlobMap()};
82  const lf::uscalfe::size_type N_dofs(dofh.NumDofs());
83  lf::assemble::COOMatrix<double> A(N_dofs, N_dofs);
84 
85  // Diffusive part of the bilinear form
86  lf::uscalfe::ReactionDiffusionElementMatrixProvider laplacian_provider(
87  fe_space_, mf_eps, lf::mesh::utils::MeshFunctionConstant(0.0));
88  lf::assemble::AssembleMatrixLocally(0, dofh, dofh, laplacian_provider, A);
89  // Convective part of the bilinear form - done based on ecu_scheme::assemble
90  if (fe_space_->LocGlobMap().NumLocalDofs(*(mesh_p->EntityByIndex(0, 0))) ==
91  6) {
92  ecu_scheme::assemble::ConvectionUpwindMatrixProvider<SCALAR,
93  decltype(velocity)>
94  convection_provider(
95  fe_space_, velocity,
96  ecu_scheme::assemble::initializeMassesQuadratic(mesh_p),
97  ecu_scheme::assemble::initializeMassesQuadraticEdges(mesh_p));
98  lf::assemble::AssembleMatrixLocally(0, dofh, dofh, convection_provider,
99  A);
100  } else if (fe_space_->LocGlobMap().NumLocalDofs(
101  *(mesh_p->EntityByIndex(0, 0))) == 3) {
102  ecu_scheme::assemble::ConvectionUpwindMatrixProvider<SCALAR,
103  decltype(velocity)>
104  convection_provider(
105  fe_space_, velocity,
106  ecu_scheme::assemble::initializeMasses(mesh_p),
107  ecu_scheme::assemble::initializeMassesQuadraticEdges(mesh_p));
108  lf::assemble::AssembleMatrixLocally(0, dofh, dofh, convection_provider,
109  A);
110  }
111 
112  // Right-hand side vector
113  Eigen::VectorXd phi(N_dofs);
114  phi.setZero();
115  lf::fe::ScalarLoadElementVectorProvider<double,
116  decltype(mf_f_test_function)>
117  load_provider(fe_space_, mf_f_test_function);
118  lf::assemble::AssembleVectorLocally(0, dofh, load_provider, phi);
119  // IMPOSE DIRICHLET CONDITIONS:
120  // Create boundary flags
121  auto bd_flags{lf::mesh::utils::flagEntitiesOnBoundary(mesh_p, 1)};
122  // Fetch flags and values for DOFs located on the boundary
123  auto ess_bdc_flags_values{lf::fe::InitEssentialConditionFromFunction(
124  *fe_space_, bd_flags, mf_g_dirichlet)};
125  // Eliminate Dirichlet dofs from linear system
126  lf::assemble::FixFlaggedSolutionComponents<double>(
127  [&ess_bdc_flags_values](lf::uscalfe::glb_idx_t gdof_idx) {
128  return ess_bdc_flags_values[gdof_idx];
129  },
130  A, phi);
131  // EnforceInflowBDCOneform(fe_space_, A, phi, dirichlet);
132 
133  // SOLVE LINEAR SYSTEM
134  Eigen::SparseMatrix<double> A_crs = A.makeSparse();
135  Eigen::SparseLU<Eigen::SparseMatrix<double>> solver;
136  solver.compute(A_crs);
137  if (solver.info() != Eigen::Success) {
138  std::cerr << "LU decomposition failed for manufactured solution"
139  << std::endl;
140  }
141  Eigen::VectorXd solution_vector = solver.solve(phi);
142  if (solver.info() != Eigen::Success) {
143  std::cerr << "Solving failed for manufactured solution" << std::endl;
144  }
145  return solution_vector;
146  } // end ComputeSolution
