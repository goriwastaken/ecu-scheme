7 std::vector<std::tuple<Eigen::Vector2d, double, bool>>
8 prepareFifteenPointQuadRule() {
9  std::vector<std::tuple<Eigen::Vector2d, double, bool>> quad_rule_result;
10  // these points are with respect to the reference triangle
11  // weights are computed with Mathematica to give a fifteen-point quadrature
12  // rule exact for degree 4 polynomials
13  std::vector<double> weights = {0.0, 4. / 45., -1.0 / 45., 4. / 45.,
14  0.0, 4. / 45., 8. / 45., 8. / 45.,
15  4. / 45., -1. / 45., 8. / 45., -1. / 45.,
16  4. / 45., 4. / 45., 0.0};
17  // flag to indicate the vertices and edge midpoints that don't correspond to
18  // S_2^0(M) nodes. different nodes are marked as true
19  std::vector<bool> flags_different_nodes = {false, true, false, true, false,
20  true, true, true, true, false,
21  true, false, true, true, false};
22  const int kMaxDegreeOfExactness = 4;
23  int temporary_index = 0;
24  for (int i = 0; i <= kMaxDegreeOfExactness; ++i) {
25  for (int j = 0; j <= kMaxDegreeOfExactness; ++j) {
26  if (i + j > kMaxDegreeOfExactness) {
27  continue;
28  }
29  Eigen::Vector2d point;
30  point << (static_cast<double>(i) /
31  static_cast<double>(kMaxDegreeOfExactness)),
32  (static_cast<double>(j) / static_cast<double>(kMaxDegreeOfExactness));
33  std::tuple point_weight_flag =
34  std::make_tuple(point, weights.at(temporary_index),
35  flags_different_nodes.at(temporary_index));
36  quad_rule_result.emplace_back(point_weight_flag);
37  temporary_index++;
38  }
39  }
40  LF_ASSERT_MSG(temporary_index == 15,
41  "Temporary index must be 15 at the end of insertions");
42 
43  return quad_rule_result;
44 }
